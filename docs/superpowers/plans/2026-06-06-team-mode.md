# Team Mode + Tie-Break Wheel Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an optional 2-team mode where the winner is each team's summed score, with a server-authoritative weighted "lucky wheel" tie-break when team sizes differ.

**Architecture:** Pure settlement/wheel logic lives in a new dependency-free `team-logic.js` (unit-tested with node's built-in test runner). `server.js` owns room/team state and emits authoritative results. `public/index.html` adds the lobby team UI, grouped leaderboard, and the animated wheel which only *plays* the server's decided result.

**Tech Stack:** Node.js, socket.io, vanilla JS + Canvas. Tests use built-in `node:test` + `node:assert` (no new deps).

```js
// Pure, dependency-free team-mode logic. Shared by server.js and tests.

// teams: { A: { count, total }, B: { count, total } }
function computeOutcome(teams) {
    const A = teams.A, B = teams.B;
    if (A.count === B.count) {
        if (A.total > B.total) return { type: 'TEAM_WIN', winner: 'A' };
        if (B.total > A.total) return { type: 'TEAM_WIN', winner: 'B' };
        return { type: 'DRAW' };
    }
    // unequal counts
    if (A.total > 0 && B.total === 0) return { type: 'TEAM_WIN', winner: 'A' };
    if (B.total > 0 && A.total === 0) return { type: 'TEAM_WIN', winner: 'B' };
    if (A.total === 0 && B.total === 0) return { type: 'WHEEL_ELIGIBLE', mode: 'fifty' };
    return { type: 'WHEEL_ELIGIBLE', mode: 'weighted' };
}

// fractions summing to 1; both-zero => 50/50
function wheelSlices(totalA, totalB) {
    let wA = totalA, wB = totalB;
    if (wA === 0 && wB === 0) { wA = 1; wB = 1; }
    const sum = wA + wB;
    return { sliceA: wA / sum, sliceB: wB / sum };
}

// rng: () => [0,1). returns 'A' | 'B'
function pickWinner(sliceA, rng) {
    return rng() < sliceA ? 'A' : 'B';
}

// Wheel arcs (degrees, 0 at top, clockwise): A = [0, sliceA*360), B = [sliceA*360, 360).
// Returns an angle strictly inside the winner arc (with a small margin off the edges).
function landAngle(winner, sliceA, rng) {
    const aEnd = sliceA * 360;
    const lo = winner === 'A' ? 0 : aEnd;
    const hi = winner === 'A' ? aEnd : 360;
    const span = hi - lo;
    const margin = Math.min(span * 0.1, 5);
    return lo + margin + rng() * (span - 2 * margin);
}

module.exports = { computeOutcome, wheelSlices, pickWinner, landAngle };
```

- [ ] **Step 1: Add spin_wheel and decide_by_score handlers**

Immediately after the `submit_score` handler's closing `});`, add:

```js
    // Host spins the tie-break wheel. Server decides the winner authoritatively.
    socket.on('spin_wheel', ({ roomId }) => {
        const room = rooms[roomId];
        if (!room || room.host !== socket.id || room.status !== 'ended') return;
        if (!room.teamMode || !room.outcome || room.outcome.type !== 'WHEEL_ELIGIBLE') return;
        if (room.wheelSpun) return;
        room.wheelSpun = true;
        const { sliceA, sliceB } = wheelSlices(room.teams.A.total, room.teams.B.total);
        const winner = pickWinner(sliceA, Math.random);
        const angle = landAngle(winner, sliceA, Math.random);
        room.finalWinner = winner;
        io.to(roomId).emit('wheel_result', { sliceA, sliceB, winner, landAngle: angle });
    });

    // Host declines the wheel and decides by raw totals (only meaningful when totals differ).
    socket.on('decide_by_score', ({ roomId }) => {
        const room = rooms[roomId];
        if (!room || room.host !== socket.id || room.status !== 'ended') return;
        if (!room.teamMode || !room.outcome || room.outcome.type !== 'WHEEL_ELIGIBLE') return;
        if (room.wheelSpun) return;
        room.wheelSpun = true;
        const a = room.teams.A.total, b = room.teams.B.total;
        const winner = a === b ? null : (a > b ? 'A' : 'B');
        room.finalWinner = winner;
        io.to(roomId).emit('team_result', { winner });
    });
```

```js
    $('wheel-controls').style.display = 'none';
    banner.style.display = 'block';
    if (!winner) { banner.innerText = 'DRAW'; banner.style.color = '#cfd3dc'; }
    else { banner.innerText = `TEAM ${winner} WINS`; banner.style.color = winner === 'A' ? '#2d7dff' : '#ff5c5c'; }
}

// Draw a 2-arc wheel rotated by `rotationDeg`. Pointer is fixed at the top (12 o'clock).
function drawWheel(sliceA, rotationDeg) {
    const c = $('wheelCanvas'), x = c.getContext('2d');
    const cx = c.width / 2, cy = c.height / 2, r = Math.min(cx, cy) - 8;
    x.clearRect(0, 0, c.width, c.height);
    const rot = rotationDeg * Math.PI / 180;
    // arc A starts at top (-90deg) going clockwise for sliceA of the circle
    const start = -Math.PI / 2 + rot;
    const aEnd = start + sliceA * 2 * Math.PI;
    const fullEnd = start + 2 * Math.PI;
    x.beginPath(); x.moveTo(cx, cy); x.arc(cx, cy, r, start, aEnd); x.closePath(); x.fillStyle = '#2d7dff'; x.fill();
    x.beginPath(); x.moveTo(cx, cy); x.arc(cx, cy, r, aEnd, fullEnd); x.closePath(); x.fillStyle = '#ff5c5c'; x.fill();
    // pointer (fixed, at top)
    x.beginPath(); x.moveTo(cx, 2); x.lineTo(cx - 10, 18); x.lineTo(cx + 10, 18); x.closePath(); x.fillStyle = '#fff'; x.fill();
}

function animateWheel(r) {
    $('wheel-controls').style.display = 'none';
    $('result-banner').innerText = 'SPINNING...';
    $('result-banner').style.color = '#ffce4f';
    const c = $('wheelCanvas'); c.style.display = 'block';
    // We want the winner arc's landAngle to end under the top pointer.
    // Pointer sits at the arc position (start + landAngleFraction). Rotating the
    // wheel by `final` places landAngle at the top: final = spins*360 - landAngle.
    const spins = 5;
    const final = spins * 360 - r.landAngle;
    const duration = 5000;
    const t0 = performance.now();
    const ease = (p) => 1 - Math.pow(1 - p, 3); // ease-out cubic: fast then slow
    function frame(now) {
        const p = Math.min(1, (now - t0) / duration);
        const rot = ease(p) * final;
        drawWheel(r.sliceA, rot);
        if (p < 1) requestAnimationFrame(frame);
        else showTeamWinner(r.winner);
    }
    requestAnimationFrame(frame);
}
```
